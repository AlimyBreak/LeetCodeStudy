class Solution
{
    public:
        int findNumbers(vector<int>& nums) 
        {
            int ret_val = 0;
            for(int i = 0 ; i < nums.size();i++)
            {
                if(nums[i]!=0)
                {
                    string s = to_string(nums[i]);
                    int length = s.size();
                    if(length % 2 == 0)
                    {
                        ret_val++;
                    }
                }
            }
            return ret_val;
        }
};
