class Solution
{
    public:
        vector<int> searchRange(vector<int>& nums, int target)
        {
            vector<int> vi(2,-1);

            if(nums.size() < 1)
            {
                return vi;
            }

            if(nums[0]== target)
            {
                vi[0] = 0;
            }
            
            if(nums[nums.size()-1]== target)
            {
                vi[1] = nums.size()-1;
            }
            
            for(int i = 1 ; i < nums.size();i++)
            {
                if(nums[i]==target && nums[i-1]!=target )
                {
                    vi[0] = i;
                }
                
                if(nums[i]!=target && nums[i-1]==target )
                {
                    vi[1] = i-1;
                    break;
                }
                
            }
            return vi;
        }
};

class Solution
{
    public:
        vector<int> searchRange(vector<int>& nums, int target)
        {
            vector<int> vi(2,-1);
            int left    =   0               ; 
            int right   =   nums.size()-1   ;
            int mid     =   0               ;
            bool flag   =   false           ;
            
            while(right >= left)
            {
                mid = left + (right - left)/2;
                if(nums[mid] ==  target)
                {
                    flag = true;
                    break;
                }
                else if(nums[mid] > target)
                {
                    right = mid - 1;
                }
                else
                {
                    left = mid + 1;
                }
            }
            
            if(flag)
            {
                left = mid;
                right = mid;
                while(left >=0 && nums[left]==target)
                {
                    left--;
                }
                
                
                while(right < nums.size() && nums[right]==target)
                {
                    right++;
                }
                
                vi[0] = left + 1;
                vi[1] = right - 1;
                
                
            }
            return vi;
        }
};
