分析:

+ 在按升序排列的数组中找插入位置，从低向高遍历即可，只要找到第一个大于等于目标数据的索引返回即可；

+ 暴力遍历的时间复杂度是$O(n)$，方法一；

+ 对于有序数组查询位置，还可以使用二分法，时间复杂度是$O(\log{(n)})$，方法二。

方法一:C_Solution,暴力遍历.

```C
int searchInsert(int* nums, int numsSize, int target)
{
    int i = 0;
    
    if(target <= nums[0])
    {
        return 0;
    }
    
    if(target > nums[numsSize-1])
    {
        return numsSize;
    }
    
    for(i=1;i<numsSize;i++)
    {
        if(target<=nums[i])
        {
            return i ;
        }
    }
    
    return -1; /* Anti warnning ,若在这里返回,那输入一定有问题.*/
}
```

方法二:C_Solution,二分法

```c
int searchInsert(int* nums, int numsSize, int target)
{
    int i           =   0;
    int ret_val     =   0;
    int left        =   0;
    int right       =   numsSize - 1;
    int mid         =   0;
    int temp1       =   0;
    
    if(target <= nums[0])
    {
        return 0;
    }
    if(target > nums[numsSize-1])
    {
        return numsSize;
    }

    while(left<=right)
    {
        mid     = left + (right-left)/2; /*防止加法溢出*/
        temp1   = nums[mid];
        if( temp1 > target)
        {
            right = mid - 1;
            continue;
        }
        else if(temp1==target)
        {
            return mid;
        }
        else
        {
            if(nums[mid+1]>=target)
            {
                return mid + 1;
            }
            else
            {
                left = mid + 1;
            }    
        }
    }
    return -1;/* Anti warnning ,若在这里返回,那输入一定有问题.*/
}
```
